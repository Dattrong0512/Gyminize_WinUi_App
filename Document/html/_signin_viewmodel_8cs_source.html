180 private async void OnLoginByGoogle()
181 {
182 // Generate state and PKCE values
183 string state = randomDataBase64url(32);
184 string codeVerifier = randomDataBase64url(32);
185 string codeChallenge = base64urlencodeNoPadding(sha256(codeVerifier));
186 const string codeChallengeMethod = "S256";
187
188 // Store the state and code_verifier values into local settings
189 ApplicationDataContainer localSettings = ApplicationData.Current.LocalSettings;
190 localSettings.Values["state"] = state;
191 localSettings.Values["code_verifier"] = codeVerifier;
192
193 // Create the OAuth 2.0 authorization request URL
194 string authorizationRequest = string.Format(
195 "{0}?response_type=code&scope=openid%20profile%20email&redirect_uri={1}&client_id={2}&state={3}&code_challenge={4}&code_challenge_method={5}",
196 authorizationEndpoint,
197 Uri.EscapeDataString(redirectURI),
198 clientID,
199 state,
200 codeChallenge,
201 codeChallengeMethod
202 );
203 // Open the authorization request URL in the browser
204 var success = await Launcher.LaunchUriAsync(new Uri(authorizationRequest));
205 if (!success)

207 await new MessageDialog("Failed to launch OAuth request.").ShowAsync();
208 }

214 private async Task ListenForOAuthCallback()
215 {
216 HttpListener listener = new HttpListener();
217 listener.Prefixes.Add("http://localhost:8080/"); // Listening on port 8080
218 listener.Start();
219
220
221 HttpListenerContext context = await listener.GetContextAsync(); // Wait for the request
222 HttpListenerRequest request = context.Request;
223 string queryString = request.Url.Query;
224
225 Dictionary<string, string> queryStringParams = queryString.Substring(1).Split('&')
226 .ToDictionary(c => c.Split('=')[0], c => Uri.UnescapeDataString(c.Split('=')[1]));
227
228 if (queryStringParams.ContainsKey("error"))
229 {
230 output($"OAuth authorization error: {queryStringParams["error"]}");
231 return;
232 }
233
234 if (!queryStringParams.ContainsKey("code") || !queryStringParams.ContainsKey("state"))
235 {
236 output("Malformed authorization response.");
237 return;
238 }
239
240 // Gets the Authorization code & state
241 string code = queryStringParams["code"];
242 string incomingState = queryStringParams["state"];
243
244 // Retrieve the expected state value from local settings (saved when the request was made)
245 ApplicationDataContainer localSettings = ApplicationData.Current.LocalSettings;
246 string expectedState = (string)localSettings.Values["state"];
247
248 if (incomingState != expectedState)
249 {
250 output($"Invalid state ({incomingState}) received.");
251 return;
252 }
253
254 // Reset the state value to avoid replay attacks
255 localSettings.Values["state"] = null;
256
257 // Authorization Code is now ready to use!
258 //output($"Authorization code: {code}");
259
260 string codeVerifier = (string)localSettings.Values["code_verifier"];
261 await ExchangeCodeForTokensAsync(code, codeVerifier); // Sử dụng mã ủy quyền để lấy token
262
263 // Send response to browser
264 HttpListenerResponse response = context.Response;
265 string responseString = "<html><body>Authentication complete. You can close this window.</body></html>";
266 byte[] buffer = Encoding.UTF8.GetBytes(responseString);
267 response.ContentLength64 = buffer.Length;
268 response.OutputStream.Write(buffer, 0, buffer.Length);
269 response.OutputStream.Close();
270
271 listener.Stop();
272 }

277 string tokenRequestBody = string.Format(

290 HttpClient client = new HttpClient(handler);

292 Debug.WriteLine("Exchanging code for tokens...");

294 string responseString = await response.Content.ReadAsStringAsync();
295 Debug.WriteLine(responseString);

444 private static string randomDataBase64url(uint length)

450 private static IBuffer sha256(string input)
451 {
452 var provider = Windows.Security.Cryptography.Core.HashAlgorithmProvider.OpenAlgorithm("SHA256");
453 var buffer = Windows.Security.Cryptography.CryptographicBuffer.ConvertStringToBinary(input, BinaryStringEncoding.Utf8);
454 return provider.HashData(buffer);
455 }

457 public static string base64urlencodeNoPadding(IBuffer buffer)
458 {
459 string base64 = CryptographicBuffer.EncodeToBase64String(buffer);
460 base64 = base64.Replace("+", "-").Replace("/", "_").Replace("=", "");
461 return base64;
